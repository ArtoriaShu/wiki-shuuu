```powershell title="install.bat"
<# ::
@echo off
net session >nul 2>&1
if %errorLevel% neq 0 (
    echo 请右键此文件，选择"以管理员身份运行"
    pause
    exit /b 1
)
PowerShell -NoProfile -ExecutionPolicy Bypass -Command "& ([scriptblock]::Create([System.IO.File]::ReadAllText('%~f0', [System.Text.Encoding]::UTF8)))"
pause
exit /b
#>

#Requires -RunAsAdministrator
<#
.SYNOPSIS
    一键下载并安装/卸载常用工具：PixPin、EcoPaste、Pot、Everything、InputTips
.PARAMETER DownloadDir
    安装包临时存放目录，默认 %TEMP%\ToolInstaller
.PARAMETER KeepInstallers
    安装完成后保留安装包
.PARAMETER GitHubToken
    GitHub Personal Access Token，用于提高 API 速率限制（未认证限 60 次/小时）
    也可通过环境变量 GITHUB_TOKEN 传入
.EXAMPLE
    .\install.ps1
    .\install.ps1 -DownloadDir "D:\Downloads"
    .\install.ps1 -KeepInstallers
    .\install.ps1 -GitHubToken "ghp_xxxx"
#>
param(
    [string]$DownloadDir = "",
    [switch]$KeepInstallers,
    [string]$GitHubToken = $env:GITHUB_TOKEN
)

$ErrorActionPreference = "Stop"
[Net.ServicePointManager]::SecurityProtocol = [Net.SecurityProtocolType]::Tls12

# 自动检测并应用系统代理
$proxyUrl = $null
try {
    $regProxy = Get-ItemProperty "HKCU:\Software\Microsoft\Windows\CurrentVersion\Internet Settings" -ErrorAction SilentlyContinue
    if ($regProxy.ProxyEnable -eq 1 -and $regProxy.ProxyServer) {
        $proxyUrl = "http://$($regProxy.ProxyServer)"
    }
} catch { }
if ($proxyUrl) {
    Write-Host "  检测到系统代理: $proxyUrl" -ForegroundColor DarkGray
} else {
    Write-Host "  未检测到系统代理，如需代理请输入（如 http://127.0.0.1:7890），直接回车跳过:" -ForegroundColor Yellow
    $input = (Read-Host "  代理地址").Trim()
    if ($input) { $proxyUrl = $input }
}
if ($proxyUrl) {
    [System.Net.WebRequest]::DefaultWebProxy = New-Object System.Net.WebProxy($proxyUrl, $true)
    [System.Net.WebRequest]::DefaultWebProxy.Credentials = [System.Net.CredentialCache]::DefaultCredentials
}

trap {
    Write-Host "`n[错误] $_" -ForegroundColor Red
    Read-Host "`n按 Enter 键退出"
    exit 1
}

function Write-Step { param([string]$m) Write-Host "`n>>> $m" -ForegroundColor Cyan }
function Write-Ok   { param([string]$m) Write-Host "    [OK] $m" -ForegroundColor Green }
function Write-Err  { param([string]$m) Write-Host "    [FAIL] $m" -ForegroundColor Red }
function Write-Info { param([string]$m) Write-Host "    $m" -ForegroundColor Gray }

# ── 菜单辅助 ──────────────────────────────────────────────────────────────────

function Show-MainMenu {
    Write-Host ""
    Write-Host "========== 工具管理器 ==========" -ForegroundColor Cyan
    Write-Host "  1. 安装工具"
    Write-Host "  2. 卸载工具"
    Write-Host "  0. 退出"
    Write-Host "================================" -ForegroundColor Cyan
}

function Show-ToolMenu {
    param([string]$Action)
    Write-Host ""
    Write-Host "--- 选择要${Action}的工具 ---" -ForegroundColor Cyan
    for ($i = 0; $i -lt $AllTools.Count; $i++) {
        Write-Host "  $($i+1). $($AllTools[$i])"
    }
    Write-Host "  A. 全部"
    Write-Host "  0. 返回"
    Write-Host ""
}

function Read-ToolSelection {
    param([string]$Action)
    Show-ToolMenu $Action
    $userInput = (Read-Host "请输入编号（多选用逗号分隔，如 1,3,5）").Trim()
    if ($userInput -eq "0") { return $null }
    if ($userInput -match '^[Aa]$') { return $AllTools }
    $selected = @()
    foreach ($part in ($userInput -split ',')) {
        $n = $part.Trim()
        if ($n -match '^\d+$') {
            $idx = [int]$n - 1
            if ($idx -ge 0 -and $idx -lt $AllTools.Count) {
                $selected += $AllTools[$idx]
            } else {
                Write-Host "  无效编号: $n，已跳过" -ForegroundColor Yellow
            }
        }
    }
    if ($selected.Count -eq 0) {
        Write-Host "  未选择任何工具" -ForegroundColor Yellow
        return $null
    }
    return $selected
}

# ── 下载辅助 ──────────────────────────────────────────────────────────────────

function Get-GitHubLatestAsset {
    param([string]$Repo, [string]$Pattern)
    Write-Info "查询 GitHub 最新版本: $Repo"
    $headers = @{ "User-Agent" = "Mozilla/5.0" }
    if ($GitHubToken) { $headers["Authorization"] = "Bearer $GitHubToken" }
    $proxyArgs = if ($proxyUrl) { @{ Proxy = $proxyUrl; ProxyUseDefaultCredentials = $true } } else { @{} }
    # 优先用 API，限速时降级到抓 HTML
    try {
        $release = Invoke-RestMethod `
            -Uri "https://api.github.com/repos/$Repo/releases/latest" `
            -Headers $headers -TimeoutSec 30 @proxyArgs
        $asset = $release.assets | Where-Object { $_.name -match $Pattern } | Select-Object -First 1
        if ($asset) {
            Write-Info "最新版本: $($release.tag_name)  文件: $($asset.name)"
            return $asset.browser_download_url
        }
    } catch { }
    # 降级：直接抓 HTML Release 页面
    Write-Info "API 不可用，改用网页抓取..."
    $allContent = ""
    try {
        $page = Invoke-WebRequest "https://github.com/$Repo/releases/latest" -UseBasicParsing -Headers $headers -TimeoutSec 30 @proxyArgs
        $allContent += $page.Content
        $tag = $page.BaseResponse.ResponseUri.ToString() -replace '.*/tag/', ''
        if ($tag) {
            try {
                $page2 = Invoke-WebRequest "https://github.com/$Repo/releases/expanded_assets/$tag" -UseBasicParsing -Headers $headers -TimeoutSec 30 @proxyArgs
                $allContent += $page2.Content
            } catch { }
        }
    } catch {
        throw "无法访问 GitHub Release 页面: $_"
    }
    $links = [regex]::Matches($allContent, "/$Repo/releases/download/[^""'<\s]+") |
             ForEach-Object { "https://github.com" + $_.Value } |
             Where-Object { $_ -match $Pattern } |
             Select-Object -Unique
    if (-not $links) { throw "在 $Repo 中未找到匹配 '$Pattern' 的下载链接" }
    $url = @($links)[0]
    Write-Info "找到链接: $url"
    return $url
}

function Get-EverythingDownloadUrl {
    Write-Info "查询 Everything 最新版本..."
    $proxyArgs = if ($proxyUrl) { @{ Proxy = $proxyUrl; ProxyUseDefaultCredentials = $true } } else { @{} }
    $page  = Invoke-WebRequest "https://www.voidtools.com/downloads/" -UseBasicParsing -TimeoutSec 30 @proxyArgs
    $match = [regex]::Match($page.Content, '/Everything-\d+\.\d+\.\d+\.\d+\.x64-Setup\.exe')
    if (-not $match.Success) { throw "无法从 voidtools.com 获取下载链接" }
    return "https://www.voidtools.com$($match.Value)"
}

function Get-PixPinDownloadUrl {
    Write-Info "查询 PixPin 最新版本..."
    $pattern = 'https://[^"''<>\s]+/PixPin[^"''<>\s]*\.exe'
    $proxyArgs = if ($proxyUrl) { @{ Proxy = $proxyUrl; ProxyUseDefaultCredentials = $true } } else { @{} }
    foreach ($site in @("https://pixpin.cn/download/", "https://pixpin.cn/", "https://pixpinapp.com/")) {
        try {
            $page  = Invoke-WebRequest $site -UseBasicParsing -TimeoutSec 15 @proxyArgs
            $match = [regex]::Match($page.Content, $pattern)
            if ($match.Success) {
                Write-Info "找到下载链接: $($match.Value)"
                return $match.Value
            }
        } catch {
            Write-Info "访问 $site 失败，尝试下一个..."
        }
    }
    throw "无法自动获取 PixPin 下载链接，请手动前往 https://pixpin.cn/download/ 下载"
}

function Save-File {
    param([string]$Url, [string]$Dest, [string]$Label = "")
    $fileName = if ($Label) { $Label } else { [System.IO.Path]::GetFileName($Dest) }
    try {
        $wc = New-Object System.Net.WebClient
        $wc.Headers.Add("User-Agent", "ToolInstaller/1.0")
        $lastPct = -1
        $wc.add_DownloadProgressChanged({
            param($s, $e)
            $pct = $e.ProgressPercentage
            if ($pct -ne $lastPct -and $pct % 5 -eq 0) {
                $script:lastPct = $pct
                $bar = "#" * [int]($pct / 5)
                $empty = "-" * (20 - [int]($pct / 5))
                $mb = [math]::Round($e.BytesReceived / 1MB, 1)
                $total = if ($e.TotalBytesToReceive -gt 0) { [math]::Round($e.TotalBytesToReceive / 1MB, 1) } else { "?" }
                Write-Host "`r    [$bar$empty] $pct%  $mb/$total MB  $fileName" -NoNewline -ForegroundColor Cyan
            }
        })
        $task = $wc.DownloadFileTaskAsync($Url, $Dest)
        while (-not $task.IsCompleted) { Start-Sleep -Milliseconds 200 }
        Write-Host ""
        if ($task.IsFaulted) { throw $task.Exception.InnerException }
        $wc.Dispose()
    } catch {
        throw "下载失败 ($fileName): $_"
    }
}

function Invoke-NsisInstaller {
    param([string]$Path, [string]$InstallDir = "")
    $args = "/S"
    if ($InstallDir) { $args = "/S /D=$InstallDir" }
    $p = Start-Process -FilePath $Path -ArgumentList $args -Wait -PassThru
    if ($p.ExitCode -ne 0) { throw "安装程序退出码: $($p.ExitCode)" }
}

function Invoke-InnoInstaller {
    param([string]$Path, [string]$InstallDir = "")
    $args = "/VERYSILENT /SUPPRESSMSGBOXES /NORESTART"
    if ($InstallDir) { $args += " /DIR=`"$InstallDir`"" }
    $p = Start-Process -FilePath $Path -ArgumentList $args -Wait -PassThru
    if ($p.ExitCode -ne 0) { throw "安装程序退出码: $($p.ExitCode)" }
}

# ── 卸载辅助 ──────────────────────────────────────────────────────────────────

function Get-UninstallEntry {
    param([string]$DisplayNamePattern)
    $paths = @(
        "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\*",
        "HKLM:\SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*",
        "HKCU:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\*"
    )
    foreach ($path in $paths) {
        $entry = Get-ItemProperty $path -ErrorAction SilentlyContinue |
                 Where-Object { $_.DisplayName -match $DisplayNamePattern } |
                 Select-Object -First 1
        if ($entry) { return $entry }
    }
    return $null
}

function Invoke-UninstallEntry {
    param(
        [string]$DisplayNamePattern,
        [string]$ExtraArgs = ""
    )
    $entry = Get-UninstallEntry $DisplayNamePattern
    if (-not $entry) { throw "未在注册表中找到已安装的程序（匹配: $DisplayNamePattern）" }
    Write-Info "找到: $($entry.DisplayName)"
    $uninstStr = $entry.UninstallString
    if ($uninstStr -match '^"?(.+?\.exe)"?\s*(.*)$') {
        $exe      = $Matches[1]
        $baseArgs = $Matches[2].Trim()
        $allArgs  = (@($baseArgs, $ExtraArgs) | Where-Object { $_ -ne "" }) -join " "
        $p = Start-Process -FilePath $exe -ArgumentList $allArgs -Wait -PassThru
        if ($p.ExitCode -ne 0) { throw "卸载程序退出码: $($p.ExitCode)" }
    } else {
        throw "无法解析卸载命令: $uninstStr"
    }
}

function Remove-InputTips {
    $dest = "$env:LOCALAPPDATA\InputTip"
    $lnk  = "$env:APPDATA\Microsoft\Windows\Start Menu\Programs\InputTip.lnk"
    Get-Process -Name "InputTip" -ErrorAction SilentlyContinue | Stop-Process -Force
    if (Test-Path $dest) { Remove-Item -Recurse -Force $dest }
    if (Test-Path $lnk)  { Remove-Item -Force $lnk }
    Write-Info "已删除 $dest 及快捷方式"
}

# ── 安装检测 ──────────────────────────────────────────────────────────────────

function Test-IsInstalled {
    param([string]$Tool)
    switch ($Tool) {
        "InputTips"  { return Test-Path "$env:LOCALAPPDATA\InputTip\InputTip.exe" }
        "EcoPaste"   { return $null -ne (Get-UninstallEntry "EcoPaste") }
        "Pot"        { return $null -ne (Get-UninstallEntry "pot") }
        "Everything" { return $null -ne (Get-UninstallEntry "Everything") }
        "PixPin"     { return $null -ne (Get-UninstallEntry "PixPin") }
    }
    return $false
}

# ── 安装执行 ──────────────────────────────────────────────────────────────────

function Get-ToolInfo {
    param([string]$Tool, [string]$DownloadDir)
    switch ($Tool) {
        "EcoPaste"   { return @{ Url = (Get-GitHubLatestAsset "EcoPasteHub/EcoPaste" "x64-setup\.exe$"); File = "$DownloadDir\EcoPaste-setup.exe"; Type = "nsis" } }
        "Pot"        { return @{ Url = (Get-GitHubLatestAsset "pot-app/pot-desktop" "_x64-setup\.exe$"); File = "$DownloadDir\pot-setup.exe"; Type = "nsis" } }
        "Everything" { return @{ Url = (Get-EverythingDownloadUrl); File = "$DownloadDir\Everything-setup.exe"; Type = "nsis" } }
        "PixPin"     { return @{ Url = (Get-PixPinDownloadUrl); File = "$DownloadDir\PixPin-setup.exe"; Type = "inno" } }
        "InputTips"  { return @{ Url = (Get-GitHubLatestAsset "abgox/InputTip" "^InputTip\.exe$"); File = "$DownloadDir\InputTip.exe"; Type = "portable" } }
    }
}

function Invoke-Install {
    param([string[]]$Tools)

    # 弹出文件夹选择对话框
    Write-Host ""
    Write-Host "  >>> 请在弹出的窗口中选择安装路径..." -ForegroundColor Cyan
    $defaultPath = "C:\Program Files"
    Add-Type -AssemblyName System.Windows.Forms
    $dialog = New-Object System.Windows.Forms.FolderBrowserDialog
    $dialog.Description = "请选择安装目录"
    $dialog.SelectedPath = $defaultPath
    $dialog.ShowNewFolderButton = $true
    if ($dialog.ShowDialog() -eq [System.Windows.Forms.DialogResult]::OK) {
        $installPath = $dialog.SelectedPath
    } else {
        $installPath = $defaultPath
        Write-Host "  未选择目录，使用默认: $defaultPath" -ForegroundColor Yellow
    }
    if (-not (Test-Path $installPath)) { New-Item -ItemType Directory -Force -Path $installPath | Out-Null }
    Write-Host "  安装目录: $installPath" -ForegroundColor DarkGray

    if (-not $DownloadDir) { $script:DownloadDir = "$env:TEMP\ToolInstaller" }
    New-Item -ItemType Directory -Force -Path $script:DownloadDir | Out-Null

    # 过滤掉已安装的工具
    $toInstall = $Tools | Where-Object {
        if (Test-IsInstalled $_) {
            Write-Host "  [跳过] $_ 已安装" -ForegroundColor Yellow
            $false
        } else { $true }
    }
    if (-not $toInstall) { return @{} }

    # 获取所有下载信息
    Write-Step "获取下载链接..."
    $toolInfos = @{}
    foreach ($tool in $toInstall) {
        try {
            $toolInfos[$tool] = Get-ToolInfo $tool $script:DownloadDir
            Write-Ok "$tool 链接获取成功"
        } catch {
            Write-Err "$tool 获取链接失败: $_"
            $toolInfos[$tool] = $null
        }
    }

    # 并行下载所有工具
    Write-Step "并行下载中..."
    $jobs = @{}   # tool -> Task
    $wcs  = @{}   # tool -> WebClient (保持引用防止 GC)
    foreach ($tool in $toInstall) {
        $info = $toolInfos[$tool]
        if (-not $info) { continue }
        $wc = [System.Net.WebClient]::new()
        $wc.Headers.Add("User-Agent", "ToolInstaller/1.0")
        if ($proxyUrl) {
            $wc.Proxy = New-Object System.Net.WebProxy($proxyUrl, $true)
            $wc.Proxy.Credentials = [System.Net.CredentialCache]::DefaultCredentials
        }
        $wcs[$tool]  = $wc
        $jobs[$tool] = $wc.DownloadFileTaskAsync([uri]$info.Url, $info.File)
    }

    # 显示并行下载进度
    $done = [System.Collections.Generic.HashSet[string]]::new()
    while ($done.Count -lt $jobs.Count) {
        $line = ""
        foreach ($tool in $jobs.Keys) {
            $task = $jobs[$tool]
            $file = $toolInfos[$tool].File
            if ($done.Contains($tool)) {
                $line += "  [$tool] 完成  "
            } elseif ($task.IsCompleted -or $task.IsFaulted -or $task.IsCanceled) {
                $mb = if (Test-Path $file) { [math]::Round((Get-Item $file).Length / 1MB, 1) } else { 0 }
                $done.Add($tool) | Out-Null
                if ($task.IsFaulted) {
                    $line += "  [$tool] 失败  "
                    $toolInfos[$tool] = $null
                } else {
                    $line += "  [$tool] ${mb}MB 完成  "
                }
            } else {
                $mb = if (Test-Path $file) { [math]::Round((Get-Item $file).Length / 1MB, 1) } else { 0 }
                $line += "  [$tool] ${mb}MB 下载中  "
            }
        }
        Write-Host "`r$line" -NoNewline -ForegroundColor Cyan
        if ($done.Count -lt $jobs.Count) { Start-Sleep -Milliseconds 500 }
    }
    Write-Host ""
    $wcs.Values | ForEach-Object { $_.Dispose() }

    # 逐个安装（PixPin 放最后）
    $results = @{}
    $installOrder = $toInstall | Where-Object { $_ -ne "PixPin" }
    if ($toInstall -contains "PixPin") { $installOrder = @($installOrder) + "PixPin" }
    foreach ($tool in $installOrder) {
        $info = $toolInfos[$tool]
        if (-not $info) { $results[$tool] = "失败: 无法获取下载链接"; continue }
        Write-Step "[$tool] 安装中..."
        try {
            if ($info.Type -eq "nsis") {
                $p = Start-Process -FilePath $info.File -ArgumentList "/S /D=$installPath\$tool" -PassThru
                if (-not $p.WaitForExit(120000)) { $p.Kill(); throw "安装超时（120秒）" }
                if ($p.ExitCode -ne 0) { throw "安装程序退出码: $($p.ExitCode)" }
            } elseif ($info.Type -eq "inno") {
                $p = Start-Process -FilePath $info.File -ArgumentList "/VERYSILENT /SUPPRESSMSGBOXES /NORESTART /DIR=`"$installPath\$tool`"" -PassThru
                if (-not $p.WaitForExit(120000)) { $p.Kill(); throw "安装超时（120秒）" }
                if ($p.ExitCode -ne 0) { throw "安装程序退出码: $($p.ExitCode)" }
            } elseif ($info.Type -eq "portable") {
                $dest = "$env:LOCALAPPDATA\InputTip\InputTip.exe"
                New-Item -ItemType Directory -Force -Path (Split-Path $dest) | Out-Null
                Copy-Item $info.File $dest -Force
                $shell = New-Object -ComObject WScript.Shell
                $lnk = $shell.CreateShortcut("$env:APPDATA\Microsoft\Windows\Start Menu\Programs\InputTip.lnk")
                $lnk.TargetPath = $dest; $lnk.Save()
                Write-Info "已安装到 $dest"
            }
            Write-Ok "$tool 安装成功"
            $results[$tool] = "成功"
        } catch {
            Write-Err "$tool 安装失败: $_"
            $results[$tool] = "失败: $_"
        }
    }

    if (-not $KeepInstallers) {
        Remove-Item -Recurse -Force $script:DownloadDir -ErrorAction SilentlyContinue
    }
    return $results
}

# ── 卸载执行 ──────────────────────────────────────────────────────────────────

function Invoke-Uninstall {
    param([string[]]$Tools)
    $results = @{}
    foreach ($tool in $Tools) {
        Write-Step "[$tool] 卸载"
        if (-not (Test-IsInstalled $tool)) {
            Write-Host "    [跳过] $tool 未检测到安装" -ForegroundColor Yellow
            $results[$tool] = "未安装（跳过）"
            continue
        }
        try {
            switch ($tool) {
                "EcoPaste"   { Invoke-UninstallEntry "EcoPaste"   $UninstallArgs["EcoPaste"] }
                "Pot"        { Invoke-UninstallEntry "pot"        $UninstallArgs["Pot"] }
                "Everything" { Invoke-UninstallEntry "Everything" $UninstallArgs["Everything"] }
                "PixPin"     { Invoke-UninstallEntry "PixPin"     $UninstallArgs["PixPin"] }
                "InputTips"  { Remove-InputTips }
            }
            Write-Ok "$tool 卸载成功"
            $results[$tool] = "成功"
        } catch {
            Write-Err "$tool 卸载失败: $_"
            $results[$tool] = "失败: $_"
        }
    }
    return $results
}
```
